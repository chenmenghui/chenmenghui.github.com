```php
<?php

class A extends B {
    public function a1()
    {
        echo __CLASS__;
    }

    public function c(){
        echo __CLASS__;
    }
}

class B{
    public function b1()
    {
        $this->c();
    }

    public function c()
    {
        echo __CLASS__;
    }
}

$a = new A();
$res = $a->b1();
echo $res; // 居然是A
```


```php
<?php

class A extends B {
    public static function a1()
    {
        echo __CLASS__;
    }

    public static function c(){
        echo __CLASS__;
    }
}

class B{
    public static function b1()
    {
        static::c();
    }

    public static function c()
    {
        echo __CLASS__;
    }
}

$res = A::b1();
echo $res; // 居然是A
```

```php

<?php
class A extends B {
    public static function a1()
    {
        echo __CLASS__;
    }

    public static function c(){
        echo __CLASS__;
    }
}

class B{
    public static function b1()
    {
        self::c();
    }

    public static function c()
    {
        echo __CLASS__;
    }
}

$res = A::b1();
echo $res; // 这才是B
```
