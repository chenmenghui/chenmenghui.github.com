```php
<?php

class A extends B {
    public function a1()
    {
        echo __CLASS__;
    }

    public function c(){
        echo __CLASS__;
    }
}

class B{
    public function b1()
    {
        $this->c();
    }

    public function c()
    {
        echo __CLASS__;
    }
}

$a = new A();
$res = $a->b1();
echo $res; // 居然是A
```
