```php
$records = array_combine(array_column($records, 'id'), $records); // 把数组中的key和其中的id对应上

foreach ($records as $key => &$value) {
    if (isset($records[$value['pid']])) {
        $records[$value['pid']]['child'][] = &$value;
    } else {
        $tree[] = &$value;
    }
}
$records = array_values($tree);
```
